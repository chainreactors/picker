```
s = '0123456789!@#$%^&*`~()_+-=[]{};:,.<>?/\'\"\\|'
for i in s:
    for j in s:
        for k in s:
            for l in s:
                for m in s:
                    #print(1)
                    md5 = hashlib.md5()
                    c = str(i) + '1_1001_10_600_600_' + str(j) + str(k) + '00' + str(l) + str(m) + '7_202103301539'
                    #print(c)
                    md5.update(c.encode('utf-8'))
                    d = md5.hexdigest()
```
